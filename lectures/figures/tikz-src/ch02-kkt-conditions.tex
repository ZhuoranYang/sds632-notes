\node[condbox, fill=lavender!12, draw=lavender] (c4) at (14.5,-4.3) {\textbf{4. Stationarity}\\[3pt] $\nabla f + \sum \lambda_i \nabla f_i + \sum \mu_j \nabla h_j = 0$};
